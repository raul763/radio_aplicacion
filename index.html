class RadioFM {

        frecuencia -= 0.1;
        if (frecuencia < 88.0) frecuencia = 108.0;

        mostrarPantalla();
    }

    void mostrarPantalla() {
        cout << "Modo: " << modo << " | Frecuencia: " << frecuencia << " MHz" << endl;
    }

    void reproducirAudio() {
        if (modo == "USB" || modo == "SD") {
            cout << "Reproduciendo audio desde " << modo << "..." << endl;
        } else {
            cout << "Escuchando se√±al " << modo << " a " << frecuencia << " MHz" << endl;
        }
    }
};

}
